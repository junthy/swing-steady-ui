@font-face {
    font-family: 'Font Awesome 5 Free';
    src: url('Font Awesome 5 Free.eot'); /* IE9*/
    src: url('Font Awesome 5 Free.eot?#iefix') format('embedded-opentype'), /* IE6-IE8 */
    url('Font Awesome 5 Free.woff') format('woff'), /* chrome、firefox */
    url('Font Awesome 5 Free.ttf') format('truetype'), /* chrome、firefox、opera、Safari, Android, iOS 4.2+*/
    url('Font Awesome 5 Free.svg#uxFont Awesome 5 Free') format('svg'); /* iOS 4.1- */
}
